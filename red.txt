  const PUBLIC_BASE_URL = "https://script.google.com/macros/s/AKfycbwemlBlhD5rpXJMUfN-VCrpYgWX4WxvT5jKRABrJ3s0zSE6wHcYVfu5qU_NBDoooRkYyw/exec";
  const MAX_TRIES = 12;
  const TRY_INTERVAL_MS = 100;

  function getBaseUrl() {
    const href = window.location.href || "";
    if (href.includes("userCodeAppPanel") || href.includes("script.googleusercontent.com")) {
      return PUBLIC_BASE_URL;
    }
    return href.split('?')[0];
  }

  function normalizeUrl(u) {
    // remove trailing slash and hash for reliable comparison
    try {
      return u.replace(/#.*$/, '').replace(/\/$/, '');
    } catch (e) { return u; }
  }

  function init() {
    const baseUrl = getBaseUrl();
    const links = document.querySelectorAll('a[href^="?page="]');
    console.debug('[NAVBAR] baseUrl:', baseUrl, 'links found:', links.length);

    if (!links || links.length === 0) {
      console.warn('[NAVBAR] Nenhum link ?page= encontrado ‚Äî aguardando elemento.');
      return false;
    }

    links.forEach(link => {
      // remove listener se j√° houver (n√£o dependemos de nomes estranhos)
      link.removeEventListener('click', link.__nav_handler__);
      // cria handler e guarda refer√™ncia para poss√≠vel remo√ß√£o depois
      const handler = handleClickWrapper.bind(link);
      link.__nav_handler__ = handler;
      link.addEventListener('click', handler);
    });

    // destaque do link atual
    const currentPage = new URLSearchParams(window.location.search).get('page') || 'index';
    links.forEach(link => {
      try {
        if (link.getAttribute('href').includes(currentPage)) {
          link.style.color = 'var(--highlight)';
          link.style.textShadow = '0 4px 18px rgba(0,0,0,0.45)';
        }
      } catch (err) { console.debug('[NAVBAR] erro ao aplicar destaque:', err); }
    });

    return true;
  }

  function tryNavigateTop(url) {
    try {
      // tentativa direta no top (padr√£o)
      window.top.location.href = url;
      console.debug('[NAVBAR] navega√ß√£o: window.top.location.href');
      return true;
    } catch (err) {
      console.debug('[NAVBAR] window.top blocked:', err && err.message ? err.message : err);
      return false;
    }
  }

  function tryOpenTop(url) {
    try {
      // window.open com target _top costuma funcionar como fallback
      const win = window.open(url, '_top');
      if (win) {
        console.debug('[NAVBAR] navega√ß√£o: window.open(..., "_top")');
        return true;
      } else {
        console.debug('[NAVBAR] window.open returned null (popup blocked?)');
        return false;
      }
    } catch (err) {
      console.debug('[NAVBAR] window.open erro:', err && err.message ? err.message : err);
      return false;
    }
  }

  function tryLocalAssign(url) {
    try {
      window.location.href = url;
      console.debug('[NAVBAR] navega√ß√£o: window.location.href (fallback)');
      return true;
    } catch (err) {
      console.debug('[NAVBAR] window.location.href erro:', err && err.message ? err.message : err);
      return false;
    }
  }

  function handleClickWrapper(e) {
    e.preventDefault();
    try {
      const baseUrl = getBaseUrl();
      const pageParam = this.getAttribute('href') || '';
      const targetUrl = baseUrl + pageParam;

      const normTarget = normalizeUrl(targetUrl);
      const normCurrent = normalizeUrl(window.location.href || '');

      if (normTarget === normCurrent) {
        console.debug('[NAVBAR] j√° estamos na mesma URL:', normTarget);
        return;
      }

      document.body.classList.add('page-exit');

      setTimeout(() => {
        console.debug('[NAVBAR] tentando navegar para:', targetUrl);

        // 1) tentativa padr√£o
        if (tryNavigateTop(targetUrl)) return;

        // 2) fallback abrir no top via window.open
        if (tryOpenTop(targetUrl)) return;

        // 3) fallback local assign (pode falhar no sandbox mas tentamos)
        if (tryLocalAssign(targetUrl)) return;

        // 4) se ainda assim nada, mostra instru√ß√£o ao usu√°rio (console + alert leve)
        console.error('[NAVBAR] todas as tentativas de navega√ß√£o falharam. URL alvo:', targetUrl);
        try {
          // informe visualmente em caso de falha intermitente
          window.alert('N√£o foi poss√≠vel navegar automaticamente. Copie e cole esta URL no navegador:\\n' + targetUrl);
        } catch (e) { /* ignore */ }
      }, 420);
    } catch (err) {
      console.error('[NAVBAR] erro n√£o esperado no clique:', err);
      try {
        const pageParam = this.getAttribute('href') || '';
        window.location.href = getBaseUrl() + pageParam;
      } catch (e) { /* swallow */ }
    }
  }

  function readyAndInit() {
    let tries = 0;
    function attempt() {
      tries++;
      const ok = init();
      if (!ok && tries < MAX_TRIES) {
        setTimeout(attempt, TRY_INTERVAL_MS);
      } else if (!ok) {
        console.error('[NAVBAR] Falha ao inicializar ap√≥s ' + tries + ' tentativas.');
      } else {
        console.info('[NAVBAR] Inicializado com sucesso (' + tries + ' tentativas).');
      }
    }

    if (document.readyState === 'complete' || document.readyState === 'interactive') {
      attempt();
    } else {
      document.addEventListener('DOMContentLoaded', attempt, { once: true });
      setTimeout(() => {
        if (document.readyState !== 'complete' && document.readyState !== 'interactive') {
          attempt();
        }
      }, 200);
    }
  }

  readyAndInit();
})();
</script>
